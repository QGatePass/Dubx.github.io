# Sales totals per region
west = customers[customers.Region=='West']
east = customers[customers.Region=='East']
central = customers[customers.Region=='Central']
south = customers[customers.Region=='South']

sales_central = central.Sales.sum().round(2)
sales_east = east.Sales.sum().round(2)
sales_west = west.Sales.sum().round(2)
sales_south = south.Sales.sum().round(2)

sales = {'west':sales_west,'east':sales_east,'central':sales_central,'south':sales_south}
sales_df = pd.Series(sales,index =sales.keys())

st = sns.countplot(data= customers,x=customers.Region,order=customers.Region.value_counts().index)
